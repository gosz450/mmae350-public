# Input user defined paramters
E_num = 200e9
sigy_num = 300e6
H_num = 1e11
delta_num = 5e-5
L_num = 1.0
A_num = 1e-4
P_num = 35000
#
# create a Sympy substitution dictionary
subs_dict = {
    E: E_num,
    sigma_y: sigy_num,
    H: H_num,
    delta: delta_num,
    L: L_num,
    A: A_num,
    P: P_num
}

fi = f.subs(subs_dict)
fpi = fp.subs(subs_dict)


# Newton's method iterations
# initial guess
ui = P_num*L_num/(E_num*A_num)
tol = 1e-8
niter = 30
residual = []
for i in range(niter):
    fi_val = fi.subs(u,ui)
    fpi_val = fpi.subs(u,ui)
    # check for convergence
    ri = abs(float(fi_val))
    residual.append(ri)
    print(ri)
    if ri < tol:
        break
    dui = -fi_val/fpi_val
    ui = ui + dui

print(residual)
print(ui)
